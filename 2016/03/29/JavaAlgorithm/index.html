插入排序算法(将指定元素插入到已经排好序的数组的合适位置)

public class InsertSort{
    public static void main(String[] args){
        int[] intArray={15,23,8,10,7};//想办法让数组变成如下的形式：{7,8,10,15,23}

        System.out.println("原始数组中元素的内容为：");
        print(intArray);

        //对数组intArray进行插入排序
        insertSort(intArray);
        System.out.println("排序数组中元素的内容为：");
        print(intArray);

    }

    static void print(int[] intArray){
        for(int item:intArray){
            System.out.print(item+"\t");
        }

        System.out.println();
    }

    /**
        对给定的数组进行插入排序

    */
    static void insertSort(int[] intArray){
        int len=intArray.length;
        for(int i=1;i<len;i++){
            for(int j=i;j>0;j--){
                if(intArray[j]<intArray[j-1]){
                    swap(intArray,j,j-1);
                }else{
                    break;
                }
            }

            //print(intArray);
        }

    }

    /**
        完成经数组中索引位i的元素的值和数组中索引值为j的元素的值进行交换位置
    */
    static void swap(int[] intArray,int i,int j){
        int temp=intArray[i];
        intArray[i]=intArray[j];
        intArray[j]=temp;
    }


}
